⚙ 모바일 동기화 설정

📱 왜 모바일에서 변경사항이 안 보이나요?

현재 데이터는 브라우저 localStorage에 저장됩니다. PC 브라우저와 모바일 브라우저는 완전히 분리된 공간이라, PC에서 저장해도 모바일에는 반영되지 않습니다.

해결책: Google Apps Script(GAS)를 DB로 연결하면 어디서든 저장·조회가 됩니다. 아래 단계를 따라 설정하세요.

📋 설정 방법 (1회만 하면 됩니다)

1

script.google.com 접속 → 새 프로젝트 만들기

2

아래 코드를 복사해서 편집기에 붙여넣기

3

배포 → 새 배포 → 유형: 웹 앱 → 액세스: 모든 사용자 → 배포

4

생성된 웹 앱 URL을 아래에 붙여넣기 후 저장

📄 Google Apps Script 코드 (복사해서 사용)

const STORE_KEY = 'shucream_v1';

function doGet(e) {
  const data = PropertiesService.getScriptProperties().getProperty(STORE_KEY);
  const parsed = data ? JSON.parse(data) : {};
  return ContentService
    .createTextOutput(JSON.stringify({ok:true, content:parsed.content||{}, images:parsed.images||{}}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    if(body.action === 'save') {
      const cur = JSON.parse(PropertiesService.getScriptProperties().getProperty(STORE_KEY)||'{}');
      if(body.content) cur.content = body.content;
      if(body.images)  cur.images  = body.images;
      PropertiesService.getScriptProperties().setProperty(STORE_KEY, JSON.stringify(cur));
      return ContentService
        .createTextOutput(JSON.stringify({ok:true}))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false, error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
